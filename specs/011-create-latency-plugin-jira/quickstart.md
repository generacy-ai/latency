# Quickstart: latency-plugin-jira

## Installation

```bash
# Install both packages
pnpm add @generacy-ai/latency-plugin-jira @generacy-ai/jira-interface
```

## Basic Usage

```typescript
import { JiraPlugin } from '@generacy-ai/latency-plugin-jira';
import type { JiraHttpAdapter } from '@generacy-ai/latency-plugin-jira';

// 1. Implement the HTTP adapter with your preferred client
const httpAdapter: JiraHttpAdapter = {
  async get<T>(path: string, params?: Record<string, string>): Promise<T> {
    const url = new URL(path, 'https://your-instance.atlassian.net');
    if (params) {
      Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));
    }
    const res = await fetch(url, {
      headers: {
        'Authorization': `Basic ${btoa(`${email}:${apiToken}`)}`,
        'Accept': 'application/json',
      },
    });
    return res.json() as T;
  },
  // ... post, put, delete similarly
};

// 2. Create the plugin
const jira = new JiraPlugin({
  http: httpAdapter,
  projectKey: 'PROJ',  // optional default project
});

// 3. Use it
const issue = await jira.getIssue('PROJ-123');
console.log(issue.key, issue.status.name);
```
